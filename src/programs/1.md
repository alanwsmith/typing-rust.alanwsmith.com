```rust, EXAMPLE1
fn main() {
  let start = 4;
  let end = 9;
  let target = 7;

  for counter in (start..=end) {
    if target > counter {
      alfa(counter);
    } else {
      bravo(counter, end);
    }
  }
}

fn alfa(value: i32) {
  println!("alfa got {}", value);

}

fn bravo(value1: i32, value2: i32) {
  println!("bravo got {} {}", value1, value2);
}
```
